Dans un repère d'origine $O$, on considère les trois points suivants, dépendant d'un nombre réel $x \in [0;10]$.
	\begin{align*}
		A  = (2;3)\cdot x && B = (-1 ; 5)\cdot(10-x) && P = (2;3) \cdot(10-x).
	\end{align*}

On admettra que les points $O, P$, et $A$ sont alignés (ils sont multiples d'un même point).

\exe{
	 Tracer dans deux repères les trois points $O, A, B,$ et $P$ : en posant $x=3$ dans le premier, et $x=8$ dans l'autre. 
}{}
\exe{
	 À l'aide de la formule de la longueur de segment vue en cours, montrer que
	 	\begin{align*}
	 		OB^2 = 26\cdot |10-x|^2, && \text{ et } && OP^2 = 13\cdot |10-x|^2.
		\end{align*}
}{}
\exe{
	Similairement, montrer que
	 	\[ BP^2 = 13 \cdot |10-x|^2.\] 
}{}
\exe{
	 Démontrer que le triangle $OBP$ est rectangle en $P$ à l'aide de la réciproque du théorème de Pythagore.
}{}
\exe{
	 En déduire que $P$ est le projeté orthogonal de $B$ sur $(OA)$ et donc que $BP$ est la hauteur du triangle $OAB$ de base $OA$.
}{}
\exe{
	 Démontrer que l'aire $\mathcal{A}(x)$ du triangle $OAB$ est donnée par, pour $x\in[0;10]$,
		\[ \mathcal{A}(x) = \dfrac{13 \cdot |10-x| \cdot |x|}2 =  \dfrac{13(10-x)x}2. \]
}{}
\exe{
	 Esquisser la courbe $\C_\mathcal{A}$ de l'aire $\mathcal{A}$, fonction réelle sur $[0;10]$.
}{}
\exe{
	 À l'aide de $\C_\mathcal{A}$, estimer la valeur du $x^\star \in [0;10]$ telle que $A(x^\star)$ soit maximale.
}{}
\exe{
	 Montrer que, pour tout $x\in[0;10]$, on a l'identité
		\[ \mathcal{A}(x) =  \dfrac{25\times13}2 - \dfrac{13}2(x-5)^2. \]
}{}
\exe{
	 En déduire que, pour tout $x\in[0;10]$,
		\[ \mathcal{A}(x) \leq \dfrac{25\times13}2, \]
	et donc que $\mathcal{A}$ atteint son maximum en $x^\star=5$.
}{}
