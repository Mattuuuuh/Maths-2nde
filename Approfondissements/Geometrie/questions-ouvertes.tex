Le but de ce document est de calculer l'aire $\mathcal{A}$ hachurée figure \ref{fig:1}.

\begin{thm}
	Pour tout $n\in\N$, 
		\[ 1 + 4 + 9 + 16 + \cdots + n^2 = \dfrac{n(n+1)(2n+1)}6. \]
\end{thm}

\begin{align*}
	A
	&\leq
	\sum_{k=1}^n \dfrac1n \dfrac{k^2}{n^2} 
	\\
	&=
	\dfrac1{n^3} \sum_{k=1}^n k^2
	\\
	&=
	\dfrac1{n^3} \dfrac{n(n+1)(2n+1)}6
	\\
	&=
	\dfrac13 + O(1/n)
\end{align*}
