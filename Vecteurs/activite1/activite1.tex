\exe{
	Dessiner les vecteurs
		\begin{align*}
			u = (-2; 4), && v = 2u, && w = -\dfrac32u, && z = -\dfrac12u,
		\end{align*}
	dans un repère.
	
	\begin{enumerate}
		\item Calculer les normes de $u, v, w,$ et $z$ et les écrire sous forme $a\sqrt{b}$ où $a, b\in\N$ et $b$ est le plus petit possible.
		\item Compléter les phrases suivantes.
			\begin{center}
				\og Lorsque le vecteur $u$ est multiplié par $\dots$ pour obtenir $v$, sa norme $\Vert u\Vert$ est multipiée par $\dots$ \fg \\ \vspace{10pt}
				\og Lorsque le vecteur $u$ est multiplié par $\dots$ pour obtenir $w$, sa norme $\Vert u\Vert$ est multipiée par $\dots$ \fg \\ \vspace{10pt}
				\og Lorsque le vecteur $u$ est multiplié par $\dots$ pour obtenir $z$, sa norme $\Vert u\Vert$ est multipiée par $\dots$ \fg
			\end{center}
		\item Pour $u = (x;y)$ et $\kappa\in\R$ quelconques, démontrer l'égalité 
			\[\Vert \kappa \cdot u \Vert = | \kappa | \cdot  \Vert u \Vert. \]
	\end{enumerate}
