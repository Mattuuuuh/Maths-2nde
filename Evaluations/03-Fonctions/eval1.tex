\begin{thm}[propriété fondamentale]\label{thm:1}
	Soit $f$ une fonction réelle sur un domaine $\D$ et $(x;y)$ un point du plan avec $x\in\D$.
	Alors
		\begin{align*}
			(x ; y) \in \C_f && \iff && \underline{\qquad} = \underline{\qquad\qquad}.
		\end{align*}
\end{thm}

\exe{2}{
	Compléter le théorème \ref{thm:1} vu en cours.
}{exe:prop-fond}{
		\begin{align*}
			(x ; y) \in \C_f && \iff && y = f(x).
		\end{align*}
}

\exe{4}{
	Un fonction $f$ admet le tableau de valeurs suivant.
		\begin{center}
		\def\arraystretch{1.2}
		\setlength\tabcolsep{20pt}
		\begin{tabular}{|c|c|c|c|c|}\hline
			$x$ & 0 & -1 & 1 & 2 \\ \hline
			$f(x)$ & 2 & 8 & 0 & 2\\ \hline
		\end{tabular}
		\end{center}
	Parmis les expressions algébriques suivantes, lesquelles peuvent et lesquelles ne peuvent pas correspondre à $f(x)$ ?
	Justifier.
		\begin{multicols}{4}
		\begin{enumerate}[label=\roman*)]
			\item $2(1-x)$
			\item $2-6x$
			\item $\dfrac{6-x}3$
			\item $2(x^2 - 2x + 1)$
		\end{enumerate}
		\end{multicols}
}{exe:f4}{
	\begin{enumerate}[label=\roman*)]
		\item 
		$2(1-x)$ est impossible car $f(-1) = 8$.
		\item 
		$2-6x$ est impossible car $f(1) = 0$.
		\item
		$\dfrac{6-x}3$ est impossible car $f(-1) = 8$.
		\item 
		$2(x^2 - 2x + 1)$ est possible car tous les images correspondent bien au tableau.
	\end{enumerate}
}


\exe{3, difficulty=1}{
	Soit $f(x) = (x+10)(x-15)$, une fonction sur $\R$.
	\begin{enumerate}
		\item Développer l'expression pour vérifier que $f(x) = x^2 - 5x - 150$.
		\item Calculer les images de 0 ; de -10 ; et de 16 par $f$ avec la forme adéquate.
		\item Comment appelle-t-on les deux différentes formes de $f$ ?
	\end{enumerate}
}{exe:image-selon-forme}{
	\begin{enumerate}
		\item
		Par double distributivité :
			\begin{align*}
				f(x) &= (x+10)(x-15) \\
					&= x^2 - 15x + 10x - 150 \\
					&= x^2 - 5x - 150
			\end{align*}
		\item
		$f(0) = -150$ avec l'une ou l'autre forme de façon immédiate.
		
		$f(-10) = 0$ grâce à la forme factorisée.
		
		$f(16) = (16+10)(16-15) = 26$, grâce à la forme factorisée.
		\item
		$f(x) = (x+10)(x-15)$ est la forme factorisée, et $f(x) = x^2 - 5x - 150$ la forme développée réduite.
	\end{enumerate}
}

\exe{2, difficulty=1}{
	Montrer, par le calcul littéral, le \emph{théorème de Léonie} ci-dessous.
	
	\emph{Indication : faire un dessin.}
}{exe:thm-Leonie}{
	La distance entre $x$ et $y$ est égale à $y-x$, car $x$ est supposé plus petit que $y$.
	En divisant par 2 et en ajoutant $x$, on trouve bien
		\[ 
		\dfrac{y-x}2 + x = \dfrac{y-x}2 + \dfrac{2x}2 = \dfrac{y-x+2x}2 = \dfrac{x+y}2,
		\]
	la moyenne de $x$ et de $y$.
}

\begin{thm}[de Léonie] \label{thm:leonie}
	Soient $x < y$ deux réels.
	Alors la moyenne de $x$ et de $y$ est égale à $x$ plus la moitié de la distance entre $x$ et $y$.
\end{thm}
