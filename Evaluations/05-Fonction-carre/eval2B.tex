\exe{4, difficulty=1}{
	Considérons la fonction $f$ sur $\R$ donnée algébriquement par
		\[ f(x) =x^4 - 6x^2 + 8. \]
	\begin{enumerate}
		\item
		Montrer que $f$ est paire.
		\item
		Montrer que $f(-2) = 0$.
		\item
		En déduire une autre solution $x\neq-2$ de l'équation $f(x) = 0$.
		\item($\star$)
		Montrer que le nombre de solutions $x\in\R$ de l'équation $f(x) = 0$ est pair.
	\end{enumerate}
}{exe:quartique2}{
	\begin{enumerate}
		\item
		En utilisant que $(-x)^n = x^n$ pour $n$ pair, on trouve
			\[ f(-x) = (-x)^4 - 6(-x)^2 + 8 = x^4 - 6x^2 + 8 = f(x) \]
		pour tout $x\in\R$.
		\item
		$f(-2) = (-2)^4 - 6\times(-2)^2 + 8 = 4^2 - 6\times4 + 8 = 16 - 24 + 8 = 0$.
		\item
		Par parité de $f$, $f(2) = f(-2) = 0$ et donc $x=2$ est une autre solution de $f(x) = 0$.
		\item($\star$)
		Par parité de $f$, les solutions se regroupent par paires de signes opposés (comme 2 et $-2$ ci-dessus).
		La seule solution pouvant potentiellement être seule est 0, mais $f(0) = 8 \neq 0$, donc il y a bien un nombre pair de solutions.
		
		D'ailleurs, $f(x) = (x+2)(x-2)\left(x+\sqrt2\right)\left(x-\sqrt2\right)$ (à vérifier !), et les seules solutions de $f(x) = 0$ sont $\pm2$ et $\pm\sqrt2$ d'après la propriété du produit nul.
	\end{enumerate}
}

\exe{4, difficulty=1}{
	Considérons la fonction $f$ sur $\R$ donnée algébriquement par
		\[ f(x) =x^4 - 6x^2 + 8. \]
	\begin{enumerate}
		\item
		Montrer que $f(x) = -1 + \left(x^2 - 3\right)^2$ pour tout $x\in\R$.
		\item
		En déduire l'extremum (maximum ou minimum) de $f$ ainsi que les antécédents qui le réalisent.
	\end{enumerate}
}{exe:quartique1}{
	\begin{enumerate}
		\item
		Pour tout $x\in\R$, et en partant de la droite,
			\[ -1 + \left(x^2 - 3\right)^2 = -1 + (x^4 + 9 - 6x^2) = x^4 - 6x^2 + 8 = f(x). \]
		\item
		Le carré est toujours positif et nul lorsque l'expression dans le carré est nulle.
		Il suit que $f(x) \geq -1$ pour tout $x\in\R$ avec égalité quand $x^2 - 3 = 0$.
		Cette dernière équation admet deux solutions, $\sqrt3$ et $-\sqrt3$, qui sont les deux antécédents réalisant le minimum $-1$.
	\end{enumerate}
}


\exe{2}{
	Est-ce que l'égalité suivante est vraie ?
	Justifier.
	\[  \sqrt{(-2)^{64}} \stackrel{?}{=} (-2)^{32} \]
}{exe:sqrt9}{
	En mettant le membre de droite au carré, on trouve
		\[ \left((-2)^{28}\right)^2 = (-2)^{28\times2} = (-2)^{56}. \]
	Donc $(-2)^{28}$ est égal à $\sqrt{(-2)^{56}}$ ou $-\sqrt{(-2)^{56}}$.
	Comme 28 est pair, $(-2)^{28}$ est positif et on a bien
		\[  \sqrt{(-2)^{56}} = (-2)^{28} \]
}
