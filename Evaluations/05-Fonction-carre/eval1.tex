\exe{5, difficulty=1}{
	Soit $f$ la fonction définie sur $\D = [-3 ; 5]$ par
		\[ f(x) = 3 - (x-1)^2. \]
	\begin{enumerate}
		\item 
		Développer et réduire l'expression algébrique de $f$.
		\item 
		Grapher la courbe $\C_f$ dans le repère figure \ref{fig:1}.
		\item
		Déterminer maximum de $f$ ainsi que l'antécédent $x^\star$ qui le réalise de deux manières différentes en décrivant votre démarche :
			\begin{enumerate}[label=\alph*)]
				\item graphiquement, à l'aide de la courbe tracée figure \ref{fig:1} ; et
				\item algébriquement, en montrant rigoureusement que $f(x) \leq f(x^\star)$ pour tout $x\in\D$.
			\end{enumerate}
		\item
		Déterminer quels $x\in\D$ vérifient $f(x) \geq -6$ de deux manières différentes en décrivant votre démarche :
			\begin{enumerate}[label=\alph*)]
				\item graphiquement, à l'aide de la courbe tracée figure \ref{fig:1} ; et
				\item algébriquement, en résolvant l'inéquation $f(x) \geq -6$ rigoureusement.
			\end{enumerate}
		%\item
		%En général, suffit-il de grapher une fonction sur un domaine $\D$ bien choisi pour connaître son maximum sur $\R$ ?
	\end{enumerate}
}{exe:canonique}{
	\begin{enumerate}
		\item 
			\[ f(x) = 3 - (x-1)^2 = 3 - (x^2 + 2x + 1) = -x^2 - 2x + 2. \]
		\item \,\\
		\begin{center}

			\end{axis}
		\end{tikzpicture}
		\end{center}
		\item
			\begin{enumerate}[label=\alph*)]
				\item 
				L'abscisse du point le plus haut de la courbe se situe en $x^\star \approx 1$.
				\item 
				$(x-1)^2 \geq 0$ pour tout $x\in\R$ avec égalité lorsque $x = 1$.
				Il suit que $f(x) \leq 3$ sous les mêmes conditions.
				Donc $f(x)$ est maximale en $x^\star = 1$.
			\end{enumerate}
		\item \,\\
		\begin{center}
		\begin{tikzpicture}[scale=.8, >=stealth]
			\begin{axis}[xmin = -3, xmax=5, ymin=-15, ymax=5, axis x line=middle, axis y line=middle, axis line style=->, grid=both, xtick distance = 1, ytick distance=2, x=50pt, y=17pt, minor y tick num = 1]%extra x ticks ={0}, extra y ticks = {0}]

			\end{axis}

				\item

				En notation mathématiques, 
					\[ \bigset{ x\in\D \tq f(x) \geq -6} \approx [-2 ; 4]. \]
				\item 
				Posons $f(x) \geq -6$ et resolvons pour $x$.

						3 - (x-1)^2 &\geq -6 \\

					\[ \bigset{ x\in\D \tq f(x) \geq -6} = [-2 ; 4]. \]
			\end{enumerate}
	\end{enumerate}

\exe{3}{
	Montrer que la fonction $f$ donnée algébriquement sur $\R$ par
		\[ f(x) = (x+3)^6 + (x-3)^6 \]
	est paire.
}{exe:f-paire}{
	Il s'agit de montrer que $f(-x) = f(x)$ pour tout $x\in\R$.
	Il est déconseillé de développer $f(x)$, même si c'est bien sûr possible.
	
	Pour tout $x\in\R$,
	\begin{align*}
		f(-x) &= (-x + 3)^6 + (-x-3)^6 && \text{ en remplaçant $x$ par $-x$} \\ 
			&= \left[ (-1)\cdot(x - 3) \right]^6 + \left[ (-1)\cdot(x+3) \right]^6 && \text{ en factorisant par $-1$} \\
			&= (-1)^6 (x-3)^6 + (-1)^6 (x+3)^6 && \text{ en utilisant $(ab)^6 = a^6b^6$ } \\
			&= (x-3)^6 + (x+3)^6 && \text{ en utilisant $(-1)^6 = +1$} \\
			&= (x+3)^6 + (x-3)^6 && \text{ en échangeant les termes de la somme }\\
			&= f(x)
	\end{align*}
}

\exe{2}{
	Est-ce que l'égalité suivante est vraie ?
	Justifier.
	\[  \sqrt{(-2)^{56}} \stackrel{?}{=} (-2)^{28} \]
}{exe:sqrt9}{
	En mettant le membre de droite au carré, on trouve
		\[ \left((-2)^{28}\right)^2 = (-2)^{28\times2} = (-2)^{56}. \]
	Donc $(-2)^{28}$ est égal à $\sqrt{(-2)^{56}}$ ou $-\sqrt{(-2)^{56}}$.
	Comme 28 est pair, $(-2)^{28}$ est positif et on a bien
		\[  \sqrt{(-2)^{56}} = (-2)^{28} \]
}


\exe{2, difficulty=2}{
	Soient $a, b > 0$ deux réels strictement positifs.
	Montrer que
		\[ \dfrac{a}{b} + \dfrac{b}a \geq 2, \]
	avec égalité uniquement lorsque $a=b$.
}{exe:Dreyer}{
	Simplifions $\frac{a}{b} + \frac{b}a - 2$ avec les propriétés des fractions.
	Le dénominateur commun étant $ab$, on calcule
		\begin{align*}
			\dfrac{a}{b} + \dfrac{b}a - 2 &= \dfrac{a^2}{ab} + \dfrac{b^2}{ab} - \dfrac{2ab}{ab}, \\
											&= \dfrac{a^2 + b^2 - 2ab}{ab}, \\
											&= \dfrac{(a-b)^2}{ab}.
		\end{align*}
	Les quantités $a, b,$ et $(a-b)^2$ sont toutes positives ou nulles, donc
		\begin{align*}
			\dfrac{a}{b} + \dfrac{b}a - 2 \geq 0 && \iff && \dfrac{a}{b} + \dfrac{b}a \geq 2
		\end{align*}
	L'égalité advient lorsque $(a-b)^2 = 0$, uniquement possible lorsque $a=b$.
	
	Nous avons donc trouvé l'extremum d'une fonction à deux variables, celui-ci étant atteint en une infinité de valeurs (la diagonale $a=b$).
}
