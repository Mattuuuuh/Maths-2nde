\subsection*{Préliminaires}

\begin{multicols}{2}

\exe{}{
	Donner les deux $x$ solutions de $f(x) = 0$.
}{exe:roots}{
	$x^2 - 7 = 0$ a pour solutions $\sqrt7$ et $-\sqrt7$ d'après le cours.
}

\exe{}{
	Esquisser la courbe représentative de $f$ dans le repère ci-contre.
	
	Où lire $\sqrt7$ dans ce repère ?
}{exe:Cf}{

	On lit $\sqrt7$ à l'intersection de $\C_f$ avec l'axe des abscisses, car $f\bigl(\sqrt7\bigr) = 0$.
}

\exe{, difficulty=0}{
	Trouver un encadrement d'amplitude 1 de $\sqrt7$ à l'aide de $\C_f$.
}{exe:balayage}{
	D'après le repère, $2 < \sqrt7 < 3$.
	En fait, $4 < 7 < 9$ implique cet encadrement car la fonction racine carrée est croissante d'après le cours.
}

\exe{, difficulty=0}{
	Trouver un encadrement d'amplitude $0,2$ de $\sqrt7$ à l'aide de $\C_f$.
}{exe:balayage2}{
	Le pas de graduation en abscisse est de $0,2$ dans le repère.
	D'après celui-ci, $2,6 < \sqrt7 < 2,8$.
	On peut vérifier que $2,6^2 = 6,76 < 7 < 7,84 = 2,8^2$, ce qui permet de conclure d'après le cours.
}

\exe{}{
	Admettons que fonction $f$ soit croissante autour de $\sqrt7$ (elle préserve l'ordre).

	Quel est le signe de $f(x)$ pour $x$ proche de $\sqrt7$ et inférieur ?
	Et pour $x$ proche de $\sqrt7$ et supérieur ?
}{exe:signesqrt7}{
	Si $x < \sqrt7$, alors $f(x) < f\bigl(\sqrt7\bigr) = 0$ est négatif.
	Si $x > \sqrt7$, alors $f(x) > f\bigl(\sqrt7\bigr) = 0$ est positif.
	
	$f(x)$ passe donc du positif au négatif en passant par 0 en $x=\sqrt7$.
	C'est cette propriété qu'on utilisera pour approximer $\sqrt7$ : en augmentant progressivement $x$ jusqu'à ce que $f(x)$ devienne positif.
}

\begin{tikzpicture}[>=stealth, scale=1.2]
\begin{axis}[xmin = 0, xmax=4, ymin=-10, ymax=10, axis x line=middle, axis y line=middle, ytick distance=5, minor tick num= 4, extra y ticks={0}]
	\addplot[transparent, thick, domain =0:4, samples=100] {x^2 - 7}  node[above left, pos=.8] {$\C_f$};
\end{axis}
\end{tikzpicture}
\end{center}
\end{multicols}

\exe{, difficulty=1}{
	Rédiger une procédure pour obtenir un encadrement de $\sqrt7$ d'amplitude $0,001$ en utilisant l'expression « tant que » et l'exercice \ref{exe:signesqrt7}.
	S'inspirer des encadrements trouvés précédemment.
}{exe:balayage3}{
	De façon analogue aux exercices précédents, partons de $x=0$ et parcourons les graduations de pas $0,001$ jusqu'à encadrer $\sqrt7$.
	
	\begin{enumerate}[label=\underline{Étape \arabic*.}, leftmargin=3cm]
		\item
		Initialiser $\ell = 0$.
		\item
		Tant que $f(\ell) < 0$, ajouter $0,001$ à $\ell$.
		\item
		Renvoyer l'encadrement
			$ \ell - 0,001 \leq \sqrt7 \leq \ell. $
	\end{enumerate}
	
	Justifions désormais que l'encadrement obtenu est bien un encadrement de $\sqrt7$ en utilisant que $f$ est croissante et donc qu'elle préserve l'ordre.
	
	Premièrement, si on avait $\ell < \sqrt7$, alors on aurait $f(\ell) < f\bigl(\sqrt7\bigr) = 0$.
	Cependant la procédure ne s'arrête que lorsque $f(\ell) \geq 0$, donc ce n'est pas possible.
	Il suit donc que $\ell < \sqrt7$ n'est pas vrai, et que $\sqrt7 \leq \ell$.
	
	Similairement, si $\ell - 0,001 > \sqrt7$, alors en appliquant $f$, on obtient que $f(\ell - 0,001) > 0$.
	Cela n'est pas possible non plus car dans ce cas, la procédure se serait arrêtée une étape avant (elle s'arrête dès que $f(\ell) \geq 0$.
	
	En conclusion, l'encadrement $\ell - 0,001 \leq \sqrt7 \leq \ell$ tient bien.
}

\subsection*{Application : le nombre d'or}

\exe{, difficulty=1}{
	On souhaite approximer la valeur du \emph{nombre d'or} $\phi$, qu'on sait unique solution positive de l'équation
		\[ x^2 - x - 1 = 0. \]
	On admet que la fonction $g(x) = x^2 - x- 1$ est croissante autour de $\phi$.
	\begin{enumerate}
		\item
		Donner un encadrement d'amplitude 1 de $\phi$.
		\item 
		Créer un programme d'approximation de $\phi$ à $10^{-5}$ près par balayage.
		\item 
		Montrer que $g(x) = \left(x - \dfrac{1+\sqrt{5}}2\right) \left(x - \dfrac{1-\sqrt{5}}2\right)$ et en déduire que $\phi = \dfrac{1+\sqrt{5}}2$.
		\item 
		Supposons que nous ayons une approximation de $\phi$ donnée par un encadrement 
			\[ a < \phi < b, \]
		d'amplitude $b-a = \ell$.
		
		En déduire un encadrement de $\sqrt{5}$ en fonction de $a$ et $b$, et son amplitude en fonction de $\ell$.
		L'encadrement de $\sqrt{5}$ déduit est-il plus ou moins ample que celui de $\phi$ ?
	\end{enumerate}
}{exe:phi}{
	\begin{enumerate}
		\item
		On calcule $g(0) = -1, g(1) = -1, g(2) = 1$, donc comme $g$ est supposée croissante, elle est négative en $x=1$, positive en $x=2$, et s'annule donc en $x=\phi$ entre ces deux valeurs.
			\[ 1 < \phi < 2. \]
		\item 
		Voir ci-dessous.
		On obtient
			\[ 1,61803 < \phi < 1,61804. \]
		\item 
		La double distributivité conclut. Comme $g(x) = 0$ est une équation produit nul, les seuls $x\in\R$ vérifiant $g(x) = 0$ sont $\frac{1+\sqrt5}2$ et $\frac{1-\sqrt5}2$. 
		La deuxième solution étant négative, $\phi$ est égale à la première.
		\item 
		Les encadrements suivants sont équivalents.
			\begin{align*}
				a < \dfrac{1+\sqrt5}2 < b \\
				2a  < 1 + \sqrt5 < 2b \\
				2a - 1 < \sqrt5 < 2b -1
			\end{align*}
		L'amplitude de ce dernier est $2b -1 - (2a - 1) = 2b - 1 -2a + 1 = 2(b-a) = 2\ell$, le double de l'amplitude de départ.
		L'encadrement de $\sqrt{5}$ est donc plus ample !
	\end{enumerate}
	
	\python{phi}
}
