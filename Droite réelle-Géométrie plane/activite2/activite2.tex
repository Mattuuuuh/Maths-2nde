\noindent Considérons $E\in\R$ et $a\in\R$ deux nombres réels, avec $a\geq0$ positif ou nul.

\section*{Résolution d'équations du type $|E| = a$}

\mth{
	Les nombres $E$ et $a$ vérifient l'équation
		\[ |E| = a \]
	si et seulement si 
	\begin{align*}
		E = \hspace{1cm} && \text{ ou } && E = \hspace{1cm}
	\end{align*}
}

\section*{Résolution d'inéquations du type $|E| \leq a$}

\mth{	
	Les nombres $E$ et $a$ vérifient l'équation
		\[ |E| \leq a \]
	si et seulement si  
		\[ E \in \hspace{4cm}\]
	c'est-à-dire que
	\begin{align*}
		\leq E \leq
	\end{align*}
}

\section*{Résolution d'inéquations du type $|E| \geq a$}

\mth{
	Les nombres $E$ et $a$ vérifient l'équation
		\[ |E| \geq a \]
	si et seulement si  
		\[ E \in \hspace{6cm}\]
	c'est-à-dire que
	\begin{align*}
		E \leq && \text{ ou } && E \geq
	\end{align*}
}
