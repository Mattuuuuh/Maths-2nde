\exe{ % pas = 1
	On considère la fonction affine $f$ sur $\R$ suivante.
		\[ f(x) = 3x -2 \]
	\begin{center}
	\setlength\tabcolsep{21pt}
	\renewcommand\a{1.8}
	\hspace{2.4cm}
	    \begin{tikzpicture}
	    	\foreach \x in {-2, ..., 2} {
			\draw[->, black, thick] (\a*\x,0) to[controls=+(-100:-.5) and +(-80:-.5)] node[above] {$+\dots$} (\a*\x+1,0);
		}
	    \end{tikzpicture}
    
	\begin{tabular}{|c|c|c|c|c|c|c|}\hline
		$x$ & -3 & -2 & -1 & 0 & 1 & 2 \\ \hline
		$f(x)$ &  & & & & & \\ \hline
	\end{tabular}
	
	\hspace{2.4cm}
	    \begin{tikzpicture}
	    	\foreach \x in {-2, ..., 2} {
			\draw[->, black, thick] (\a*\x,0) to[controls=+(-80:.5) and +(-100:.5)] node[below=5pt] {$+\dots$} (\a*\x+1,0);
		}
	    \end{tikzpicture}
	\end{center}
	
	\begin{enumerate}
		\item Compléter le tableau de valeurs.
		\item Est-ce un tableau de proportionnalité ?
		\item Compléter la phrase suivante.
			\begin{center}
				\og Quand $x$ augmente de $\dots$, $f(x)$ augmente de $\dots$. \fg
			\end{center}
	\end{enumerate}
}{}
