		\[ f(x) = \dfrac{(4x-8)(x-3)}{-2x^2 + 5x + 12}. \]
	\begin{enumerate}

		\item En déduire $\D_f$.
		\item Donner $\{ x \in \D_f \text{ tq. } f(x) \leq 0 \}$ sous forme d'union d'intervalles.
	\end{enumerate}
