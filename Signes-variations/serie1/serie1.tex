\exe{
	On souhaite connaître les solutions de l'équation du deuxième degré d'inconnue $x\in\R$ :
		\[ 21 x^2 - 11 x - 2  = 0. \]
}{}

\exe{
	Donner une fonction $f$ sur $\R$ non identiquement nulle telle qu'elle admette \emph{exactement} $2, -3, \dfrac27$, et $0$ pour racines.
}{}

\exe{\label{ex:1}
	Pour chacune des fonctions affines $f$ suivantes, donner l'intervalle 
		\[ \{ x \in \R \text{ tq. } f(x) \geq 0 \}. \]
	
	\begin{multicols}{2}
	\begin{enumerate}
		\item $f(x) = x+3$
		\item $f(x) = 3x-10$
		\item $f(x) = x$
		\item $f(x) = -4x-12$
		\item $f(x) = -10x$
		\item $f(x) = 1$
	\end{enumerate}
	\end{multicols}
}{}

\exe{
	À l'aide de l'exercice \ref{ex:1} remplir les tableaux de signes ci-dessous et donner
		\[ \{ x \in \R \text{ tq. } f(x) \geq 0 \} \]
	sous forme d'union d'intervalles.
	
	\begin{tikzpicture}
		\tkzTabInit
		 %[lgt=3,espcl=1.5]
		 [lgt=5]
	       		{$x$ / 1 , $x+3$ / 1, $3x-10$ / 1 , $f(x) = (x+3)(3x-10)$ / 1}
	       		{,,,}
	\end{tikzpicture}
	
	\begin{tikzpicture}
		\tkzTabInit
		 %[lgt=3,espcl=1.5]
		 [lgt=5]
	       		{$x$ / 1 , $x$ / 1, $-4x-12$ / 1 , $f(x) = x(-4x-12)$ / 1}
	       		{,,,}
	\end{tikzpicture}
	
	
	\begin{tikzpicture}
		\tkzTabInit
		 %[lgt=3,espcl=1.5]
		 [lgt=5]

	       		{,,,,}
	\end{tikzpicture}
}{}


\exe{
	On souhaite connaître pour quels $x\in\D$ l'inégalité suivante est vérifée.
		\[ f(x) = \dfrac{3-x}{6x^2 - 5x - 4} \geq 0. \]
	
	\begin{enumerate}
		\item Montrer que $6x^2 - 5x - 4 = (3x-4)(2x+1)$.
		\item En déduire les valeurs interdites et donc $\D$.
		\item Remplir le tableau de signes ci-dessous.
		\item Exprimer $\{ x \in \D \text{ tq. } f(x) \geq 0 \}$ sous forme d'union d'intervalles.
	\end{enumerate}
	
	
	\begin{tikzpicture}
		\tkzTabInit
		 %[lgt=3,espcl=1.5]
	       		{$x$ / 1, $3-x$ / 1 , $3x-4$ / 1, $2x+1$ / 1 , $f(x)$ / 1}
	       		{,,,,}
	\end{tikzpicture}
}{}
