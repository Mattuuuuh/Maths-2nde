\exe{
	On pose l'intervalle perforé $I=]3; \pinfty[-\{17\}$.
	
	\begin{enumerate}
		\item Montrer que $3x+5 > 17 - x$ sur $I$.
		\item Montrer que $7x-5 > 2(x+5)$ sur $I$.
		\item En déduire que, pour tout $x\in I$,
			\[ \dfrac{(3x+5)(7x-5)}{(17-x)(x+5)} > 2. \]
	\end{enumerate}
}{}
