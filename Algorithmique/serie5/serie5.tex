def MOYENNE(X):

def VARIANCE(X):
	moyenne = MOYENNE(X)
	Y = [(x - moyenne)**2 for x in X]
	return MOYENNE(Y)

def ECARTTYPE(X):
	return math.sqrt(VARIANCE(X))
