\begin{mintedbox}{python}

	print('COND est vraie')
else: # sinon,
	print('COND est fausse')

	print('a est strictement supérieur à b ou 2a est supérieur ou égal à b')

# si a < b ET 2a > b

	print('a < b < 2a')
