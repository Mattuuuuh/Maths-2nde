\subsection{Identités remarquables}

\framedelayed{
	Développer et réduire la forme factorisée suivante.
	\boxAB{
		$(x+3)^2$
	}{
		$(x-2)^2$
	}
}{
	\begin{align*}
		(a+b)^2 &= a^2 + b^2 + 2·a·b \\
		(a-b)^2 &= a^2 + b^2 - 2·a·b
	\end{align*}
	\boxAB{
		$(x+3)^2 = x^2 + 3^2 + 2\times x \times 3 = x^2 + 9 + 6x$
	}{
		$(x-2)^2 = x^2 + 2^2 - 2 \times x \times 2 = x^2 + 4 - 4x$
	}
}


\framedelayed{
	Développer et réduire la forme factorisée suivante.
	\boxAB{
		$(x-4)^2$
	}{
		$(x+5)^2$
	}
}{
	\begin{align*}
		(a+b)^2 &= a^2 + b^2 + 2·a·b \\
		(a-b)^2 &= a^2 + b^2 - 2·a·b
	\end{align*}
	\boxAB{
		$(x-4)^2 = x^2 + 4^2 - 2 \times x \times 4 = x^2 + 16 - 8x$
	}{
		$(x+5)^2 = x^2+ 5^2 + 2 \times x \times 5 = x^2 + 25 + 10x$
	}
}

\subsection{Images}

\framedelayed{
	Soit $f(x) = 3 + 2x^2$ une fonction.
	\boxAB{
		Calculer $f(-1)$.
	}{
		 Calculer $f(-2)$.
	}
}{
	\[ f(x) = 3 + 2x^2 \]
	\boxAB{
		$f(-1) = 3+2(-1)^2 = 3+ 2 = 5$
	}{
		$f(-2) = 3 + 2(-2)^2 = 3+2\times4 = 3+8=11$
	}
}

\subsection{Équation linéaire}

\framedelayed{
	Résoudre l'équation linéaire suivante pour $x\in\R$.
	\boxAB{
		$3x + 1 = 0$
	}{
		 $6x - 1 = 0$
	}
}{
	\boxAB{
		$3x+1=0 \iff 3x = -1 \iff x = -\dfrac13$
	}{
		 $6x-1=0 \iff 6x = 1 \iff x = \dfrac16$
	}
}
