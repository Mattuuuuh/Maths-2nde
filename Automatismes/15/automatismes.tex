\subsection{Inégalités}

\framedelayed{
	Donner l'intervalle des nombres $x\in\R$ vérifiant
	\boxAB{
		$x \leq 7$
	}{
		$x \geq 2$
	}
}{
	\boxAB{
		$x \leq 7 \iff x \in ]\minfty ; 7]$
	}{
		$x \geq 2 \iff x \in [2 ; \pinfty[$
	}
}

\subsection{Valeurs absolues}

\framedelayed{
	Calculer la valeur absolue suivante.
	\boxAB{
		$\left|-4-7\right|$
	}{
		$\left|-8-6\right|$
	}
}{
	\boxAB{
		$\abs{-4-7} = \abs{-11} = 11$
	}{
		$\abs{-8-6} = \abs{-14}  = 14$
	}
}

\framedelayed{
	Calculer la valeur absolue suivante.
	\boxAB{
		$\left|\dfrac23 - 1\right|$
	}{
		$\left|\dfrac13 - 2\right|$
	}
}{
	\boxAB{
		$\left|\dfrac23 - 1\right| = \left|\dfrac23 - \dfrac33\right| = \left|\dfrac{-1}3\right| = \dfrac13$
	}{
		$\left|\dfrac13 - 2\right| = \left|\dfrac13 - \dfrac63\right| = \left|\dfrac{-5}3\right| = \dfrac53$
	}
}
