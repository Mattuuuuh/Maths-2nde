\subsection{Coefficient multiplicateur}

\framedelayed{
	Donner le coefficient multiplicateur associé à
	\boxAB{
		une augmentation de 35\%.
	}{
		une diminution de 15\%.
	}
}{
	\begin{align*}
		+P\% &\iff \times \Bigl( 1 + \dfrac{P}{100} \Bigr) \\
		-P\% &\iff \times \Bigl( 1 - \dfrac{P}{100} \Bigr) 
	\end{align*}
	\boxAB{
		augmenter de 35\% $\iff \times 1,35$.
	}{
		diminuer de 15\% $\iff \times 0,85$.
	}
}

\subsection{Taux d'évolution}

\framedelayed{
	Donner le taux d'évolution (augmentation, diminution, \%) associé à la multiplication par
	\boxAB{
		$0,82$
	}{
		 $1,31$
	}
}{
	\[ CM-1 = 100 (CM - 1)\% \]
	\boxAB{
		$0,82 - 1 = -0,18 = -18\%$ : diminution de 18\%.
	}{
		 $1,31 - 1 = 0,31 = 31\%$ : augmentation de 31\%.
	}
}
