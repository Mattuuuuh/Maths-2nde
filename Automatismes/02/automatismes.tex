\subsection{Ensembles}

\framedelayed{
	Considérons les ensembles 
		\begin{align*}
			E &= \bigset{ 2 ; 3 ; 10 ; -2}, \\
			F &= \bigset{2 ; 10 ; 11 ; -2 ; 14}, \\
			G &= \bigset{ -2 ; 2 ; 10}.
		\end{align*}
	\boxAB{
		A-t-on $E \subseteq F$ ou $E \not\subseteq F$ ?
	}{
		A-t-on $G \subseteq E$ ou $G \not\subseteq E$ ?
	}
	\vspace{-10pt}
}{
	\boxAB{
		$E \not\subseteq F$
	}{
		$G \subseteq E$
	}
}

\subsection{Nombres décimaux}

\framedelayed{
	Donner le développement décimal de la fraction suivante.
	\boxAB{
		$\dfrac{11}{5}$
	}{
		$\dfrac{3}{20}$
	}
}{
	\boxAB{
		$\dfrac{11}{5} = \dfrac{22}{10} = 2,2$
	}{
		$\dfrac{3}{20} = \dfrac{15}{100} = 0,15$
	}
}


\framedelayed{
	Écrire le nombre suivant en écriture scientifique.
	\boxAB{
		5 370 010
	}{
		0,023 008 1
	}
}{
	\boxAB{
		$5 370 010 = 5,37001 \times 10^6$
	}{
		$0,023 008 1 = 2,30081 \times 10^{-2}$
	}
}


\framedelayed{
