\subsection{Ensembles}

\begin{frame}{1\vspace{-32pt}}
	Considérons les ensembles 
		\begin{align*}
			E &= \bigset{ 2 ; 3 ; 10 ; -2}, \\
			F &= \bigset{2 ; 10 ; 11 ; -2 ; 14}, \\
			G &= \bigset{ -2 ; 2 ; 10}.
		\end{align*}
	\boxAB{
		A-t-on $E \subseteq F$ ou $E \not\subseteq F$ ?
	}{
		A-t-on $G \subseteq E$ ou $G \not\subseteq E$ ?
	}
	\vspace{-10pt}
\end{frame}
