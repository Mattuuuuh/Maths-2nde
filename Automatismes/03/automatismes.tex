\subsection{Nombres décimaux}

\framedelayed{
	Écrire le nombre suivant en écriture scientifique.
	\boxAB{
		0,008 004 1
	}{
		26 280 700
	}
}{
	\boxAB{
		$0,008 004 1 = 8,0041\times10^{-3}$

	}
}


\subsection{Fractions}

\framedelayed{
	Exprimer le produit suivant sous forme de fraction d'entiers $\frac{a}{b}$.
	\boxAB{
		$1 + \dfrac56$
	}{
		$10 \times \dfrac{1}{7}$
	}
}{
	\boxAB{
		$\dfrac66 + \dfrac56 = \dfrac{11}6$
	}{
		$\dfrac{10}1 \times \dfrac{1}{7} = \dfrac{10}{7}$
	}
}

\framedelayed{
	Exprimer la somme suivante sous forme de fraction d'entiers $\frac{a}{b}$.
	\boxAB{
		$10 \times \dfrac{1}{6}$
	}{
		$1 + \dfrac23$
	}
}{
	\boxAB{
		$\dfrac{10}1 \times \dfrac{1}{6} = \dfrac{10}{6}$
	}{
		$\dfrac33 + \dfrac23 = \dfrac53$
	}
}

\framedelayed{
	Exprimer la somme suivante sous forme de fraction d'entiers $\frac{a}{b}$.

		$5 + \dfrac23$

		$3 + \dfrac56$
	}
}{
	\boxAB{
		$\dfrac{15}3 + \dfrac23 = \dfrac{17}3$
	}{
		$\dfrac{18}6 + \dfrac56 = \dfrac{23}6$
	}
}
