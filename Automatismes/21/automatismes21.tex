\subsection{Racines}

\begin{frame}{1}	
	Donner la racine de la fonction affine $f$. C'est le $x\in\R$ tel que $f(x) = 0$.
	\boxAB{
		\[ f(x) = \dfrac23 x - 2. \]
	}{
		\[ f(x) = -\dfrac65x - 3. \]
	}
\end{frame}

\begin{frame}{2}
	Donner la racine de la fonction affine $g$.
	\boxAB{
		\[ g(x) = \dfrac12 - 2x. \]
	}{
		\[ g(x) = \dfrac23 + x. \]
	}
\end{frame}
