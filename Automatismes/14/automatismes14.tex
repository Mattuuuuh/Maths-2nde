\subsection{Images}

\begin{frame}{2}
	Soient $f,g$ deux fonctions affines sur $\R$ données algébriquement par
		\begin{align*}
			f(x) = \dfrac23 x - 3, && \text{ et } && g(x) = -8 x + 1.
		\end{align*}
	
	\boxAB{
		Calculer $f(-3)$.
	}{
		Calculer $g(3)$.
	}
\end{frame}

\begin{frame}{3}
	Soient $F, G$ deux fonctions affines sur $\R$ données algébriquement par
		\begin{align*}
			F(x) = -x, && \text{ et } && G(x) =  \dfrac75 x +1.
		\end{align*}
	
	\boxAB{
		Calculer $F(4)$.
	}{
		Calculer $G(-5)$.
	}
\end{frame}
