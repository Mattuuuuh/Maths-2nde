\subsection{Coordonnées de vecteurs}

\begin{frame}{1}	
	Calculer les coordonnées du vecteurs $\overrightarrow{AB}$, où
	\boxAB{
		\begin{align*}
			A = (-3; 4), && B = (2 ; 3).
		\end{align*}
	}{
		\begin{align*}
			A = (7; -4), && B = (5; -2).
		\end{align*}
	}
\end{frame}

\subsection{Norme vectorielle}

\begin{frame}{2}

	}{
		\[ u = \begin{pmatrix} 3  \\ -4 \end{pmatrix}. \]
	}
\end{frame}
