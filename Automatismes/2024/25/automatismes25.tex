\subsection{Racines}

\begin{frame}{1}	
	Donner une fonction $f$ non identiquement nulle telle que
	\boxAB{
		$f$ s'annule en $5$ et en $-2$.
	}{
		$f$ s'annule en $4$ et en $-3$.
	}

\end{frame}

\subsection{Racines}

\begin{frame}{2}	
	Donner le $x\in\R$ réel vérifiant
	\boxAB{
			\[ 5-3x = 0. \]
	}{
			\[ 11-8x = 0. \]
	}
\end{frame}


\subsection{Développement}

\begin{frame}{3}	
	Développer le produit pour obtenir une expression de la forme $ax^2 + bx + c$ où $a, b, c \in\R$ sont des nombres réels.
	\boxAB{
			\[ (x - 3)\left(x-\dfrac16\right) \]
	}{
			\[ (x + 2)\left(x - \dfrac34\right) \]
	}
\end{frame}

\subsection{Domaine de définition}

	\boxAB{
			\[ f(x) = \sqrt{3+x}. \]
	}{
			\[ f(x) = \sqrt{-3+x}. \]
	}
\end{frame}
