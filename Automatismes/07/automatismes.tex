\subsection{Fonctions}

\framedelayed{
	Considérons la fonction $f$ donnée algébriquement par
		\[ f(x) = 2 + 4x. \]
	\boxAB{
		Calculer $f(-2)$.
	}{
		Calculer $f(3)$.
	}
}{
		\[ f(x) = 2 + 4x. \]
	\boxAB{
		$f(-2) = 2 + 4 \times (-2) = 2 - 8 = -6$
	}{
		$f(3) = 2 + 4 \times (3) = 2 + 12 = 14$
	}
}

\framedelayed{
	\vspace{10pt}
	Considérons la fonction $f$ donnée algébriquement par
		\[ f(x) = 2 + 6x. \]
	\boxAB{
		Calculer $f\left(\dfrac56\right)$.
	}{
		Calculer $f\left(\dfrac{-1}6\right)$.
	}
}{
		\[ f(x) = 2 + 6x. \]
	\boxAB{
		$f\left(\dfrac56\right) = 2 + 6 \times \dfrac56 = 2 + 5 = 7$
	}{
		$f\left(\dfrac{-1}6\right) = 2 + 6 \times \dfrac{-1}6 = 2 - 1 = 1$
	}
}

\framedelayed{
	Considérons la fonction $g$ donnée algébriquement par
		\[ g(x) = (2x+1)(x-1). \]
	\boxAB{
		Calculer $g(2)$.
	}{
		Calculer $g(1)$.
	}
}{
		\[ g(x) = (2x+1)(x-1). \]
	\boxAB{
		$g(2) = (2 \times 2 + 1)\times(2-1) = (5)\times(1) = 5$
	}{
		$g(1) = (2 \times 1 + 1)\times(1-1) = (3)\times(0) = 0$
	}
}
