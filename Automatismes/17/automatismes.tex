	\boxAB{
		$u =\pvec{-3}{2}$ \quad et \quad $v = \pvec{6}{-5}.$

		$2u + v = \pvec{2\times2 + (-1)}{2\times(-5)+5} = \pvec{3}{-5}$
	}
}
