\subsection{Équation linéaire}

\framedelayed{
	Résoudre l'équation suivante pour $x\in\R$.
	\boxAB{
		$\dfrac13 x + 1 = 0$
	}{
		$\dfrac{x+6}3 = 0$
	}
}{
	\boxAB{
		$\dfrac13 x + 1 = 0 \iff \dfrac13 x = -1 \iff x = -3$
	}{
		$\dfrac{x+6}3 = 0 \iff x+6 = 0 \iff x= -6$
	}
}

\subsection{Fractions}

\framedelayed{
		\[ f(x) = \dfrac13 - \dfrac12 x. \]
	Donner l'image suivante sous forme de fraction.
	\boxAB{
		 $f\left(\dfrac12\right)$
	}{
		 $f\left(\dfrac13\right)$
	}
}{
		\[ f(x) = \dfrac13 - \dfrac12 x. \]
	\boxAB{
		 $f\left(\dfrac12\right) = \dfrac13 - \dfrac12 \times \dfrac12 = \dfrac13 - \dfrac14 = \dfrac4{12} - \dfrac3{12} = \dfrac1{12}$
	}{
		$f\left(\dfrac13\right) = \dfrac13 - \dfrac12 \times \dfrac13 = \dfrac13 - \dfrac16 = \dfrac26 - \dfrac16 = \dfrac16$
	}
}
