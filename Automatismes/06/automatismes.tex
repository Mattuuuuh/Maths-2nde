\subsection{Fonctions}

\framedelayed{
	Considérons la fonction $f$ donnée algébriquement par
		\[ f(x) = 3x + 1. \]
	\boxAB{
		Donner l'image de $2$ par $f$.
	}{
		Donner l'image de $-3$ par $f$.
	}
}{
		\[ f(x) = 3x + 1. \]
	\boxAB{
		$f(2) = 3\times2 + 1 = 6 + 1 = 7.$
	}{
		$f(-3) = 3\times(-3) + 1 = -9 + 1 = -8.$
	}
}

\framedelayed{
	Considérons la fonction $f$ donnée algébriquement par
		\[ f(x) = 3x + 1. \]
	\boxAB{
		Donner l'image de $-2$ par $f$.
	}{
		Donner l'image de $3$ par $f$.
	}
}{
		\[ f(x) = 3x + 1. \]
	\boxAB{
		$f(-2) = 3\times(-2) + 1 = -6 + 1 = -5.$
	}{
		$f(3) = 3\times3 + 1 = 9 + 1 = 10.$
	}
}

\framedelayed{
	Considérons la fonction $g$ donnée algébriquement par
		\[ g(x) = 12x + 4. \]
	\boxAB{
		Donner l'image de $\frac{5}{12}$ par $g$.
	}{
		Donner l'image de $\frac{3}{12}$ par $g$.
	}
}{
		\[ g(x) = 12x + 4. \]
	\boxAB{
		$g(\frac5{12}) = 12\times\frac5{12} + 4 = 5 + 4 = 9.$
	}{
		$g(\frac3{12}) = 12\times\frac3{12} + 4 = 3 + 4 = 7.$
	}
}

\subsection{Fractions}

\framedelayed{
	Donner le résultat du calcul suivant.
	\boxAB{
		$8 + \dfrac13$
	}{
		$6 + \dfrac14$
	}
}{
	\boxAB{
		$8 + \dfrac13 = \dfrac{8 \times 3}{3} + \dfrac13 = \dfrac{24+1}3 = \dfrac{25}3.$
	}{
		$6 + \dfrac14 = \dfrac{6 \times 4}{4} + \dfrac14 = \dfrac{24+1}4 = \dfrac{25}{4}.$
	}
}
