\subsection{Fonctions}

\framedelayed{
	Considérons la fonction $f$ donnée algébriquement par
		\[ f(x) = 3 - x. \]
	\boxAB{
		Calculer $f(-5,3)$.
	}{
		Calculer $f(-4,8)$.
	}
}{
		\[ f(x) = 3 - x. \]
	\boxAB{
		$f(-5,3) = 3 - (-5,3) = 3 + 5,3 = 8,3$
	}{
		$f(-4,8) = 3 - (-4,8) = 3 + 4,8 = 7,8$
	}
}

\subsection{Intervalles}

\framedelayed{
	Compléter les pointillés avec le signe $\in$ ou $\not\in$.
	\boxAB{
		$ 3 ~~\dots~~ [-7 ; 2] $
	}{
		$ 7 ~~\dots~~ [-1 ; 10] $
	}
}{
	\boxAB{
		$ 3 ~~\not\in~~ [-7 ; 2] $
	}{
		$ 7 ~~\in~~ [-1 ; 10] $
	}
}

\subsection{Équation linéaire}

\framedelayed{
	Résoudre l'équation suivante pour $x\in\R$.
	\boxAB{
		$\dfrac{x+6}3 = 0$
	}{
		$\dfrac13 x + 1 = 0$
	}
}{
	\boxAB{
		$\dfrac{x+6}3 = 0 \iff x+6 = 0 \iff x= -6$
	}{
		$\dfrac13 x + 1 = 0 \iff \dfrac13 x = -1 \iff x = -3$
	}
}

\subsection{Fractions}

\framedelayed{
	Donner le résultat du calcul suivant sous forme de fraction.
	\boxAB{
		 $\dfrac13 - \dfrac12 \times \dfrac12$
	}{
		$\dfrac12 \times \dfrac12 - \dfrac15$
	}
}{
	\boxAB{
		 $\dfrac13 - \dfrac12 \times \dfrac12 = \dfrac13 - \dfrac14 = \dfrac4{12} - \dfrac3{12} = \dfrac1{12}$
	}{
		$\dfrac12 \times \dfrac12 - \dfrac15 = \dfrac14 - \dfrac15 = \dfrac5{20} - \dfrac4{20} = \dfrac1{20}$
	}
}
