\subsection{Images}

\framedelayed{
	Soient 
		\begin{align*}
			f(x) = 4-x && \et && g(x) = 5-x
		\end{align*}
	deux fonctions sur $\R$.
	\boxAB{
		Calculer $f(-10)$.
	}{
		Calculer $g(-5)$.
	}
}{
		\begin{align*}
			f(x) = 4-x && \et && g(x) = 5-x
		\end{align*}
	\boxAB{
		$f(-10) = 4-(-10) = 4+10 = 14.$
	}{
		$g(-5) = 5-(-5) = 5+5 = 10.$
	}
}


\subsection{Courbe représentative}

\framedelayed{
	Soient 
		\begin{align*}
			f(x) = 2x+1 && \et && g(x) = x+2
		\end{align*}
	deux fonctions sur $\R$.
	\boxAB{
		Donner un point appartenant à $\C_f$.
	}{
		Donner un point appartenant à $\C_g$.
	}
}{
		\begin{align*}
			f(x) = 2x+1 && \et && g(x) = x+2
		\end{align*}
	\boxAB{
		$f(0) = 1$ donc $(0 ; 1)$ appartient à $\C_f$.
	}{
		$g(12) = 14$ donc $(12 ; 14)$ appartient à $\C_g$.
	}
}
