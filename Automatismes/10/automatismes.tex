\subsection{Proportions}

\framedelayed{
	Dans une classe de Seconde,
		\begin{itemize}
			\item
			21 élèves se sont déjà connectés sur Éléa ; et
			\item
			14 ne se sont jamais connectés.
		\end{itemize}
	\boxAB{
		Calculer le pourcentage d'élèves s'étant déjà connectés.
	}{
		Calculer le pourcentage d'élèves ne s'étant jamais connectés.
	}
}{
	\begin{itemize}
		\item
		21 élèves se sont déjà connectés sur Éléa
		\item
		14 ne se sont jamais connectés
	\end{itemize}
	Total : $21+14 = 35$
	\boxAB{
		$\dfrac{\text{déjà connectés}}{\text{total}} = \dfrac{21}{35} = 0,6 = 60\%$.
	}{
		$\dfrac{\text{jamais connectés}}{\text{total}} = \dfrac{14}{35} = 0,4 = 40\%$.
	}
}

\subsection{Pourcentages}

\framedelayed{
	Calculer
	\boxAB{
		75\% de 40.
	}{
		125\% de 40.
	}
}{
	\boxAB{
		75\% de 40 $= \dfrac{75}{100} \times 40 = 30$.
	}{
		125\% de 40 $= \dfrac{125}{100} \times 40 = 50$.
	}
}

\subsection{Développement réduction}

\framedelayed{
	Développer réduire les produits suivants.
	\boxAB{
		 $\dfrac12 (x-2)(x-3)$
	}{
		 $\dfrac12 (x-4)(x-5)$
	}
}{
	\vspace{10pt}
	\boxAB{
		\vspace{-12pt}
		 \begin{align*}
		 \dfrac12 (x-2)(x-3) &= \dfrac12 \bigl[ x^2 - 2x - 3x + (-2)(-3) \bigr] 
		 					\\ &= \dfrac12 \bigl(x^2 - 5x + 6 \bigr) = \dfrac12x^2 - \dfrac52 x + 3
		\end{align*}
	}{
		\vspace{-12pt}
		 \begin{align*}
		 \dfrac12 (x-4)(x-5) &= \dfrac12 \bigl[ x^2 - 4x - 5x + (-4)(-5) \bigr] 
		 					\\ &= \dfrac12 \bigl(x^2 - 9x +20 \bigr) = \dfrac12x^2 - \dfrac92 x + 10
		\end{align*}
	}
}

\subsection{Équation linéaire}

\framedelayed{
	Résoudre l'équation suivante pour $x\in\R$.
	Une valeur exacte est attendue.
	\boxAB{
		$3x+4 = 0$
	}{
		$6x+7 = 0$
	}
}{
	\boxAB{
		$3x+4 = 0 \iff 3x = -4 \iff x = -\dfrac43$.
	}{
		$6x+7 = 0 \iff 6x = -7 \iff x = -\dfrac76.$
	}
	Les valeurs numériques ne sont pas exactes !
}
